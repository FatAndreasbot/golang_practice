package main

import "fmt"

type account struct {
	value int
}

func main() {
	s1 := make([]account, 0, 2) // Создается постой массив, вместимостью 2
	s1 = append(s1, account{})  // Добавляется пустая структура в список s1
	s2 := append(s1, account{}) // В массив s1 кладется доп элемент. Копия не создается,
	// тк не превышена вместимость. Этот же список кладется в слайс s2
	acc := &s2[0] // указатель на первый элемент массива
	// (обоих списков, ибо в списке хранятся только указатели)
	acc.value = 100            // обновление значения структуры
	fmt.Println(s1, s2)        // вывод - [{100}] [{100}, {0}]
	s1 = append(s1, account{}) // добавляется новая структура в первый список
	acc.value += 100           // меняется значение общего элемента
	fmt.Println(s1, s2)        // вывод - [{200}, {0}] [{200}, {0}]
}
