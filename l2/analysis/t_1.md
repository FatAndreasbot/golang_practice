```go
package main

import (
	"fmt"
)

type MyError struct {
	data string
}

func (m *MyError) Error() string {
	return m.data
}

func main() {
	err := foo(4)
	if err != nil {
		fmt.Println("oops")
	} else {
		fmt.Println("ok")
	}
}

func foo(i int) error {
	var err *MyError
	if i > 5 {
		err = &MyError{data: "i>5"}
	}
	return err
}
```
## Вывод
```
oops
```
## Объяснение
`foo()` всегда возваращает значение, поэтому `main()` никогда не 
выведет `ok`
