# Main
```go
package main

import (
	"errors"
	"fmt"
)

func main() {
	println("Case 1")
	case1()
	println()
	println()

	println("Case 2")
	case2()
	println()
	println()

	println("Case 3")
	case3()
	println()
	println()

}
```
# Case 1
```go
func case1() {
	helperWithDefer := func(isError bool) error {
		var retVal error
		defer func() {
			retVal = errors.New("Extra error")
		}()

		if isError {
			retVal = errors.New("Default error")
		}

		return retVal
	}

	helperWithoutDefer := func(isError bool) error {
		var retVal error

		if isError {
			retVal = errors.New("Default error")
		}

		return retVal
	}

	fmt.Println("\twithout:")
	fmt.Println(helperWithoutDefer(false))
	fmt.Println(helperWithoutDefer(true))
	fmt.Println("\twith:")
	fmt.Println(helperWithDefer(false))
	fmt.Println(helperWithDefer(true))
}
```
## Вывод
```
Case 1
	without:
<nil>
Default error
	with:
<nil>
Default error
```
## Объяснение
Значение `reVal` записалось в адрес вывода,
до того как отложенные команды были запущены

# Case 2

```go
func case2() {
	helperWithDefer := func(isError bool) (retVal error) {
		defer func() {
			retVal = errors.New("Extra error")
		}()

		if isError {
			retVal = errors.New("Default error")
		}
		return
	}
	helperWithoutDefer := func(isError bool) (retVal error) {
		if isError {
			retVal = errors.New("Default error")
		}

		return
	}
	fmt.Println("\twithout:")
	fmt.Println(helperWithoutDefer(false))
	fmt.Println(helperWithoutDefer(true))
	fmt.Println("\twith:")
	fmt.Println(helperWithDefer(false))
	fmt.Println(helperWithDefer(true))
}
```
## Вывод
```
Case 2
	without:
<nil>
Default error
	with:
Extra error
Extra error
```
## Объяснение
Здесь же используется одно и то же значение `reVal`
используется для значение вывода, поэтому отложенный
вызов поменял выводимое функцией значение

# Case 3
```go
func case3() {
	helperWithDefer := func(isError bool) (retVal error) {
		defer func() {
			retVal = errors.New("First Error")
		}()
		defer func() {
			retVal = errors.New("Second Error")
		}()
		if isError {
			retVal = errors.New("Default error")
		}
		return
	}

	helperWithoutDefer := func(isError bool) (retVal error) {
		if isError {
			retVal = errors.New("Default error")
		}
		return
	}

	fmt.Println("\twithout:")
	fmt.Println(helperWithoutDefer(false))
	fmt.Println(helperWithoutDefer(true))
	fmt.Println("\twith:")
	fmt.Println(helperWithDefer(false))
	fmt.Println(helperWithDefer(true))
}
```
## Вывод
```
Case 3
	without:
<nil>
Default error
	with:
First Error
First Error
```
## Объяснение
Тут дважды выводится `First Error` из-за того, что первая 
отложенная функция выполнилось последней, так как порядок 
отложенных функций выполняются по принципу `LIFO`
